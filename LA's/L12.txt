class Toy:
    def __init__(self, name, price):
        self.name = name
        self.price = price
       
    def describeToy(self):
        print(f"My toy name is {self.name} and it has a price of {self.price}.")
       
class Car(Toy):
    def __init__(self, name, price):
        super().__init__(name, price)
       
toy = Toy("Doll", "1000")
toy.describeToy()
